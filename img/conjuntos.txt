
conj_str = input("Conjunto: ")
dec = int(input("Subconjunto: "))
subn = dec
backwards = []
binary = []
conjunto = []
sub = []

for i in conj_str:
    conjunto.append(i)
print(conjunto)

while (dec / 2 >= 1):
    backwards.append(dec%2)
    dec = int(dec / 2)
backwards.append(dec%2)

while len(conjunto) > len(backwards):
    backwards.append(0)

size = len(backwards)-1
for i in range(size, -1, -1):
    binary.append(backwards[i])

for i in range(0, size):
    if binary[i] == 1:
        sub.append(conjunto[i])

print(binary)

print(f"O {subn}º subconjunto é {sub}")
